"Permission denied" on database file:
  Remove-Item .tmp/data.db
  # Restart app - file recreated with correct permissions

"Cannot find Ollama":
  ollama serve
  # Keep this running in a terminal

"Connection refused":
  Make sure you're in: src/cofounder_agent
  Check port 8000: netstat -ano | findstr :8000

"Module not found":
  Make sure .env file exists:
  Get-Content .env
